import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['SimHei', 'Microsoft YaHei', 'DejaVu Sans']
plt.rcParams['axes.unicode_minus'] = False
import seaborn as sns
from math import pi

def create_algorithm_comparison_analysis():
    # 数据准备
    algorithms = ['逻辑回归', 'K近邻(KNN)', '支持向量机(SVM)', '梯度提升机(XGBoost)', '神经网络(MLP)', '随机森林']
    
    # 性能指标数据
    data = {
        '算法': algorithms,
        '准确率': [0.98, 0.965, 0.9825, 0.985, 0.982, 0.976],
        '召回率': [0.9688, 0.932, 0.9683, 0.9219, 1.0, 0.9584],
        '精确率': [1.0, 0.967, 0.9815, 1.0, 0.952, 0.986],
        'F1值': [0.984, 0.949, 0.975, 0.959, 0.976, 0.96],
        'AUC值': [0.98, 0.98, 0.99, 0.997, 0.996, 0.99]
    }
    
    df = pd.DataFrame(data)
    
    # 创建2x3布局的清晰可视化
    fig, axes = plt.subplots(2, 3, figsize=(18, 12))
    
    # 定义颜色方案
    algo_colors = ['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FECA57', '#DDA0DD']
    metrics = ['准确率', '召回率', '精确率', 'F1值', 'AUC值']
    
    # 1. 综合性能对比柱状图 (所有6个算法)
    ax1 = axes[0, 0]
    x = np.arange(len(algorithms))
    width = 0.15
    
    metric_colors = ['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FECA57']
    
    for i, metric in enumerate(metrics):
        offset = (i - 2) * width
        bars = ax1.bar(x + offset, df[metric], width, label=metric, 
                      color=metric_colors[i], alpha=0.8, edgecolor='black', linewidth=0.5)
    
    ax1.set_xlabel('算法', fontsize=11, fontweight='bold')
    ax1.set_ylabel('性能指标', fontsize=11, fontweight='bold')
    ax1.set_title('综合性能对比', fontsize=12, fontweight='bold')
    ax1.set_xticks(x)
    ax1.set_xticklabels(algorithms, rotation=45, ha='right', fontsize=9)
    ax1.legend(bbox_to_anchor=(1.05, 1), loc='upper left', fontsize=8)
    ax1.grid(axis='y', alpha=0.3)
    ax1.set_ylim(0.9, 1.02)
    
    # 2. 雷达图 (所有6个算法)
    ax2 = axes[0, 1]
    ax2.remove()
    ax2 = fig.add_subplot(2, 3, 2, projection='polar')
    
    # 为雷达图准备角度
    angles = [n / float(len(metrics)) * 2 * pi for n in range(len(metrics))]
    angles += angles[:1]  # 闭合图形
    
    for i, algo in enumerate(algorithms):
        values = df[df['算法'] == algo][metrics].values.flatten().tolist()
        values += values[:1]  # 闭合图形
        
        ax2.plot(angles, values, 'o-', linewidth=2, label=algo, color=algo_colors[i], alpha=0.8)
        ax2.fill(angles, values, alpha=0.1, color=algo_colors[i])
    
    ax2.set_xticks(angles[:-1])
    ax2.set_xticklabels(metrics, fontsize=10)
    ax2.set_ylim(0.9, 1.0)
    ax2.set_title('雷达图性能对比', fontsize=12, fontweight='bold', pad=20)
    ax2.legend(bbox_to_anchor=(1.3, 1.0), fontsize=8)
    
    # 3. 性能热力图
    ax3 = axes[0, 2]
    heatmap_data = df.set_index('算法')[metrics].T
    sns.heatmap(heatmap_data, annot=True, cmap='YlOrRd', fmt='.3f', 
                cbar_kws={'label': '性能分数'}, ax=ax3, 
                annot_kws={'fontsize': 8})
    ax3.set_title('性能热力图', fontsize=12, fontweight='bold')
    ax3.set_xticklabels(algorithms, rotation=45, ha='right', fontsize=9)
    ax3.set_yticklabels(metrics, fontsize=9)
    
    # 4. 准确率vs召回率散点图 (所有6个算法)
    ax4 = axes[1, 0]
    scatter = ax4.scatter(df['准确率'], df['召回率'], s=df['AUC值']*300, 
                         c=algo_colors, alpha=0.7, edgecolors='black', linewidth=1)
    
    for i, algo in enumerate(algorithms):
        ax4.annotate(algo, (df['准确率'][i], df['召回率'][i]), 
                    xytext=(5, 5), textcoords='offset points', fontsize=8, fontweight='bold')
    
    ax4.set_xlabel('准确率', fontsize=11, fontweight='bold')
    ax4.set_ylabel('召回率', fontsize=11, fontweight='bold')
    ax4.set_title('准确率 vs 召回率\n(气泡大小=AUC值)', fontsize=12, fontweight='bold')
    ax4.grid(True, alpha=0.3)
    ax4.set_xlim(0.94, 1.01)
    ax4.set_ylim(0.91, 1.01)
    
    # 5. 综合评分排名 (所有6个算法)
    ax5 = axes[1, 1]
    
    # 计算综合评分（加权平均）
    weights = {'准确率': 0.25, '召回率': 0.2, '精确率': 0.2, 'F1值': 0.2, 'AUC值': 0.15}
    
    comprehensive_scores = []
    for i in range(len(algorithms)):
        score = sum(df.iloc[i][metric] * weight for metric, weight in weights.items())
        comprehensive_scores.append(score)
    
    df['综合评分'] = comprehensive_scores
    
    # 排序并绘制
    sorted_indices = np.argsort(comprehensive_scores)[::-1]  # 降序排列
    sorted_algos = [algorithms[i] for i in sorted_indices]
    sorted_scores = [comprehensive_scores[i] for i in sorted_indices]
    sorted_colors = [algo_colors[i] for i in sorted_indices]
    
    bars = ax5.barh(range(len(sorted_algos)), sorted_scores, 
                   color=sorted_colors, alpha=0.7, edgecolor='black')
    ax5.set_yticks(range(len(sorted_algos)))
    ax5.set_yticklabels(sorted_algos, fontsize=10)
    ax5.set_xlabel('综合评分', fontsize=11, fontweight='bold')
    ax5.set_title('综合评分排名', fontsize=12, fontweight='bold')
    ax5.invert_yaxis()
    
    # 添加分数标签
    for i, (bar, score) in enumerate(zip(bars, sorted_scores)):
        ax5.text(bar.get_width() + 0.001, bar.get_y() + bar.get_height()/2,
                f'{score:.4f}', ha='left', va='center', fontweight='bold', fontsize=9)
    
    ax5.grid(axis='x', alpha=0.3)
    ax5.set_xlim(0.96, 0.99)
    
    # 6. 各指标最优算法统计
    ax6 = axes[1, 2]
    
    # 统计每个算法在各指标上的表现
    best_counts = {algo: 0 for algo in algorithms}
    top2_counts = {algo: 0 for algo in algorithms}
    
    for metric in metrics:
        sorted_by_metric = df.sort_values(metric, ascending=False)
        # 最佳
        best_algo = sorted_by_metric.iloc[0]['算法']
        best_counts[best_algo] += 1
        # 前两名
        for i in range(min(2, len(sorted_by_metric))):
            top2_algo = sorted_by_metric.iloc[i]['算法']
            top2_counts[top2_algo] += 0.5
    
    algo_names = list(best_counts.keys())
    best_values = list(best_counts.values())
    top2_values = list(top2_counts.values())
    
    x_pos = np.arange(len(algo_names))
    width = 0.35
    
    bars1 = ax6.bar(x_pos - width/2, best_values, width, label='最佳次数', 
                   color='gold', alpha=0.8, edgecolor='black')
    bars2 = ax6.bar(x_pos + width/2, top2_values, width, label='前二次数', 
                   color='silver', alpha=0.8, edgecolor='black')
    
    ax6.set_xlabel('算法', fontsize=11, fontweight='bold')
    ax6.set_ylabel('次数', fontsize=11, fontweight='bold')
    ax6.set_title('各指标优势统计', fontsize=12, fontweight='bold')
    ax6.set_xticks(x_pos)
    ax6.set_xticklabels(algo_names, rotation=45, ha='right', fontsize=9)
    ax6.legend(fontsize=9)
    ax6.grid(axis='y', alpha=0.3)
    
    # 添加数值标签
    for bars in [bars1, bars2]:
        for bar in bars:
            height = bar.get_height()
            if height > 0:
                ax6.text(bar.get_x() + bar.get_width()/2., height + 0.05,
                        f'{height:.1f}', ha='center', va='bottom', fontweight='bold', fontsize=8)
    
    plt.suptitle('机器学习算法性能综合对比分析', fontsize=16, fontweight='bold')
    plt.tight_layout()
    plt.savefig('algorithm_performance_comparison.png', dpi=300, bbox_inches='tight')
    plt.show()
    
    return df

def generate_detailed_analysis(df):
    """生成详细的分析报告"""
    print("=" * 80)
    print("机器学习算法性能对比分析报告")
    print("=" * 80)
    
    algorithms = df['算法'].tolist()
    metrics = ['准确率', '召回率', '精确率', 'F1值', 'AUC值']
    
    # 1. 整体性能分析
    print("\n【1. 整体性能分析】")
    print("-" * 40)
    
    # 找出各项指标的最佳算法
    for metric in metrics:
        best_algo = df.loc[df[metric].idxmax(), '算法']
        best_score = df[metric].max()
        worst_algo = df.loc[df[metric].idxmin(), '算法']
        worst_score = df[metric].min()
        print(f"{metric:>6}: 最优={best_algo}({best_score:.4f}) | 最低={worst_algo}({worst_score:.4f})")
    
    # 2. 算法综合排名
    print("\n【2. 算法综合排名】")
    print("-" * 40)
    
    # 计算综合评分
    weights = {'准确率': 0.25, '召回率': 0.2, '精确率': 0.2, 'F1值': 0.2, 'AUC值': 0.15}
    comprehensive_scores = []
    for i in range(len(algorithms)):
        score = sum(df.iloc[i][metric] * weight for metric, weight in weights.items())
        comprehensive_scores.append(score)
    
    # 排序
    ranking_data = list(zip(algorithms, comprehensive_scores))
    ranking_data.sort(key=lambda x: x[1], reverse=True)
    
    for i, (algo, score) in enumerate(ranking_data):
        print(f"第{i+1}名: {algo:<15} 综合评分: {score:.4f}")
    
    # 3. 性能特色分析
    print("\n【3. 性能特色分析】")
    print("-" * 40)
    
    analysis_dict = {
        '逻辑回归': '精确率完美(100%)，模型简单且可解释性强，综合排名第一',
        'K近邻(KNN)': '各项指标相对较低，但模型简单易理解，适合小数据集',
        '支持向量机(SVM)': '各项指标均衡且优秀，是很好的通用分类器',
        '梯度提升机(XGBoost)': 'AUC值最高(0.997)，在复杂数据处理方面表现出色',
        '神经网络(MLP)': '召回率完美(100%)，在识别所有正例方面表现最佳',
        '随机森林': '性能稳定均衡，具有很好的特征重要性分析能力'
    }
    
    for algo, analysis in analysis_dict.items():
        print(f"• {algo}: {analysis}")
    
    # 4. 应用场景推荐
    print("\n【4. 应用场景推荐】")
    print("-" * 40)
    
    recommendations = {
        '医疗诊断(高召回率)': '神经网络(MLP) - 召回率100%',
        '金融风控(高精确率)': '逻辑回归/XGBoost - 精确率100%',
        '综合性能需求': '逻辑回归 - 综合评分最高',
        '模型可解释性': '逻辑回归 - 线性模型易解释',
        '复杂数据处理': 'XGBoost - AUC值最高',
        '特征分析需求': '随机森林 - 提供特征重要性'
    }
    
    for scenario, recommendation in recommendations.items():
        print(f"• {scenario:>12}: {recommendation}")
    
    # 5. 性能统计摘要
    print("\n【5. 性能统计摘要】")
    print("-" * 40)
    
    for metric in metrics:
        mean_score = df[metric].mean()
        std_score = df[metric].std()
        max_score = df[metric].max()
        min_score = df[metric].min()
        range_score = max_score - min_score
        print(f"{metric:>6}: 均值={mean_score:.4f}±{std_score:.4f} | 极差={range_score:.4f}")
    
    print("\n" + "=" * 80)
    print("总结：逻辑回归综合表现最优，神经网络在召回率上无可匹敌，")
    print("XGBoost在AUC值上领先，各算法都有其独特优势和适用场景。")
    print("=" * 80)

# 主函数
def main():
    """主执行函数"""
    # 创建可视化分析
    df = create_algorithm_comparison_analysis()
    
    # 生成详细分析报告
    generate_detailed_analysis(df)
    

if __name__ == "__main__":
    main()

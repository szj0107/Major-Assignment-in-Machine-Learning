# 导入必要的库
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split, cross_val_score, GridSearchCV, learning_curve, KFold
from sklearn.preprocessing import StandardScaler, LabelEncoder
from sklearn.metrics import accuracy_score, confusion_matrix, roc_curve, auc, classification_report, ConfusionMatrixDisplay, RocCurveDisplay
from sklearn.ensemble import GradientBoostingClassifier
import time
import warnings
warnings.filterwarnings('ignore')

# 设置 Matplotlib 支持中文显示
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# --- 1. 数据加载和初步探索 ---
print("--- 1. 数据加载和初步探索 ---")
df = pd.read_csv('data.csv')

# 移除不需要的列
print("\n移除 'id' 和 'Unnamed: 32' 列...")
if 'Unnamed: 32' in df.columns:
    df = df.drop('Unnamed: 32', axis=1)
if 'id' in df.columns:
    df = df.drop('id', axis=1)

# --- 2. 数据预处理 ---
print("\n--- 2. 数据预处理 ---")
target_column = 'diagnosis'
X = df.drop(target_column, axis=1)
y = df[target_column]

# 对目标变量进行编码
print(f"\n对目标变量 '{target_column}' 进行标签编码...")
le = LabelEncoder()
y = le.fit_transform(y)
print(f"编码映射: {dict(zip(le.classes_, le.transform(le.classes_)))}") # M=1, B=0

# 划分训练集和测试集
print("\n划分数据为训练集 (70%) 和测试集 (30%)...")
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42, stratify=y)

# 特征缩放
print("\n使用 StandardScaler 对数值特征进行标准化...")
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train)
X_test_scaled = scaler.transform(X_test)
X_train_scaled = pd.DataFrame(X_train_scaled, columns=X_train.columns, index=X_train.index)
X_test_scaled = pd.DataFrame(X_test_scaled, columns=X_test.columns, index=X_test.index)

# --- 梯度提升机模型 ---
print("\n--- 梯度提升机模型 ---")
model_name = "梯度提升机"
start_time = time.time()

# 定义参数网格
param_grid = {
    'n_estimators': [100, 200],
    'learning_rate': [0.05, 0.1, 0.2],
    'max_depth': [3, 5]
}

# 定义交叉验证策略
cv_strategy = KFold(n_splits=5, shuffle=True, random_state=42)

# 创建梯度提升分类器
gbm = GradientBoostingClassifier(random_state=42)

# 网格搜索
print(f"--- {model_name}: 超参数调优 ---")
grid_search = GridSearchCV(
    estimator=gbm,
    param_grid=param_grid,
    cv=cv_strategy,
    scoring='roc_auc',
    n_jobs=-1,
    verbose=1
)
grid_search.fit(X_train_scaled, y_train)

print(f"找到的最佳参数: {grid_search.best_params_}")
print(f"训练集交叉验证最佳 ROC AUC 分数: {grid_search.best_score_:.4f}")

# 获取最佳模型
best_gbm = grid_search.best_estimator_

# 绘制超参数调优过程图
try:
    results_df = pd.DataFrame(grid_search.cv_results_)
    key_param = 'learning_rate'
    plt.figure(figsize=(10, 6))
    sns.lineplot(data=results_df, x=f'param_{key_param}', y='mean_test_score', 
                 hue='param_max_depth', style='param_n_estimators', 
                 markers=True, dashes=False)
    plt.title(f'{model_name}: 超参数调优 (学习率 vs ROC AUC)\n作者：陈嘉慧-2023217459')
    plt.xlabel('学习率')
    plt.ylabel('平均 ROC AUC (交叉验证)')
    plt.grid(True)
    plt.tight_layout()
    plt.savefig('gbm_hyperparameter_tuning_陈嘉慧-2023217459.png')
    print("已保存图表: gbm_hyperparameter_tuning_陈嘉慧-2023217459.png")
    plt.show()
except Exception as e:
    print(f"无法生成超参数调优图: {e}")

# 交叉验证评估
print(f"--- {model_name}: 交叉验证性能 ---")
cv_scores = cross_val_score(
    best_gbm,
    X_train_scaled,
    y_train,
    cv=cv_strategy,
    scoring='roc_auc',
    n_jobs=-1
)
print(f"交叉验证 ROC AUC 分数: {cv_scores}")
print(f"平均交叉验证 ROC AUC: {np.mean(cv_scores):.4f} (+/- {np.std(cv_scores):.4f})")

# 绘制交叉验证结果
plt.figure(figsize=(8, 5))
sns.boxplot(x=cv_scores)
plt.title(f'{model_name}: 交叉验证 ROC AUC 分数 (5折)\n作者：陈嘉慧-2023217459')
plt.xlabel('ROC AUC 分数')
plt.tight_layout()
plt.savefig('gbm_cross_validation_陈嘉慧-2023217459.png')
print("已保存图表: gbm_cross_validation_陈嘉慧-2023217459.png")
plt.show()

# 测试集评估
print(f"--- {model_name}: 在测试集上评估 ---")
y_pred = best_gbm.predict(X_test_scaled)
y_pred_proba = best_gbm.predict_proba(X_test_scaled)[:, 1]

# 计算指标
accuracy = accuracy_score(y_test, y_pred)
cm = confusion_matrix(y_test, y_pred)
fpr, tpr, thresholds = roc_curve(y_test, y_pred_proba)
roc_auc = auc(fpr, tpr)

print(f"测试集准确率: {accuracy:.4f}")
print(f"测试集 ROC AUC: {roc_auc:.4f}")
print("分类报告:")
target_names = [f"类别 {cls} ({'良性' if cls == 0 else '恶性'})" for cls in le.transform(le.classes_)]
print(classification_report(y_test, y_pred, target_names=target_names))

# 绘制混淆矩阵
plt.figure(figsize=(8, 6))
display_labels = le.classes_
disp = ConfusionMatrixDisplay(confusion_matrix=cm, display_labels=display_labels)
disp.plot(cmap='Blues', values_format='d')
plt.title(f'{model_name}: 混淆矩阵 (测试集)\n作者：陈嘉慧-2023217459')
plt.tight_layout()
plt.savefig('gbm_confusion_matrix_陈嘉慧-2023217459.png')
print("已保存图表: gbm_confusion_matrix_陈嘉慧-2023217459.png")
plt.show()

# 绘制 ROC 曲线
plt.figure(figsize=(8, 6))
plt.plot(fpr, tpr, color='darkorange', lw=2, label=f'ROC 曲线 (面积 = {roc_auc:.3f})')
plt.plot([0, 1], [0, 1], color='navy', lw=2, linestyle='--', label='随机猜测')
plt.xlim([0.0, 1.0])
plt.ylim([0.0, 1.05])
plt.xlabel('假正例率 (False Positive Rate)')
plt.ylabel('真正例率 (True Positive Rate)')
plt.title(f'{model_name}: ROC 曲线 (测试集)\n作者：陈嘉慧-2023217459')
plt.legend(loc="lower right")
plt.grid(True)
plt.tight_layout()
plt.savefig('gbm_roc_curve_陈嘉慧-2023217459.png')
print("已保存图表: gbm_roc_curve_陈嘉慧-2023217459.png")
plt.show()

# 特征重要性
print(f"--- {model_name}: 特征重要性 ---")
importances = best_gbm.feature_importances_
feature_importance_df = pd.DataFrame({'特征': X.columns, '重要性': importances})
feature_importance_df = feature_importance_df.sort_values(by='重要性', ascending=False)

# 绘制特征重要性
plt.figure(figsize=(10, 8))
sns.barplot(x='重要性', y='特征', data=feature_importance_df.head(20))
plt.title(f'{model_name}: 特征重要性 (前20个特征)\n作者：陈嘉慧-2023217459')
plt.tight_layout()
plt.savefig('gbm_feature_importance_陈嘉慧-2023217459.png')
print("已保存图表: gbm_feature_importance_陈嘉慧-2023217459.png")
plt.show()

# 学习曲线
print(f"--- {model_name}: 学习曲线 ---")
try:
    train_sizes, train_scores, test_scores = learning_curve(
        estimator=best_gbm,
        X=X_train_scaled,
        y=y_train,
        cv=cv_strategy,
        n_jobs=-1,
        train_sizes=np.linspace(0.1, 1.0, 10),
        scoring='roc_auc'
    )

    train_scores_mean = np.mean(train_scores, axis=1)
    train_scores_std = np.std(train_scores, axis=1)
    test_scores_mean = np.mean(test_scores, axis=1)
    test_scores_std = np.std(test_scores, axis=1)

    plt.figure(figsize=(10, 6))
    plt.fill_between(train_sizes, train_scores_mean - train_scores_std,
                     train_scores_mean + train_scores_std, alpha=0.1, color="r")
    plt.fill_between(train_sizes, test_scores_mean - test_scores_std,
                     test_scores_mean + test_scores_std, alpha=0.1, color="g")
    plt.plot(train_sizes, train_scores_mean, 'o-', color="r", label="训练得分")
    plt.plot(train_sizes, test_scores_mean, 'o-', color="g", label="交叉验证得分")

    plt.title(f'{model_name}: 学习曲线\n作者：陈嘉慧-2023217459')
    plt.xlabel("训练样本数量")
    plt.ylabel("ROC AUC 分数")
    plt.legend(loc="best")
    plt.grid(True)
    plt.tight_layout()
    plt.savefig('gbm_learning_curve_陈嘉慧-2023217459.png')
    print("已保存图表: gbm_learning_curve_陈嘉慧-2023217459.png")
    plt.show()
except Exception as e:
    print(f"无法生成学习曲线: {e}")

end_time = time.time()
print(f"{model_name} 模型处理耗时: {end_time - start_time:.2f} 秒")

# --- 结果汇总 ---
print("\n--- 梯度提升机模型性能汇总 ---")
performance_df = pd.DataFrame({
    '指标': ['平均交叉验证 ROC AUC', '交叉验证 ROC AUC 标准差', '测试集准确率', '测试集 ROC AUC'],
    '值': [np.mean(cv_scores), np.std(cv_scores), accuracy, roc_auc]
})
print(performance_df.to_string(index=False, float_format="%.4f"))

print("\n--- 最重要的10个特征 ---")
print(feature_importance_df.head(10).to_string(index=False))

print("\n--- 分析完成 ---")
print("已生成的图表文件:")
print("- gbm_hyperparameter_tuning_陈嘉慧-2023217459.png")
print("- gbm_cross_validation_陈嘉慧-2023217459.png")
print("- gbm_confusion_matrix_陈嘉慧-2023217459.png")
print("- gbm_roc_curve_陈嘉慧-2023217459.png")
print("- gbm_feature_importance_陈嘉慧-2023217459.png")
print("- gbm_learning_curve_陈嘉慧-2023217459.png")
